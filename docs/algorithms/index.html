    selection sort # 

    public class Selection {
    public static void sort(Comparable[] a){
        int len = a.length;
        for (int i=0;i<len;i++){
            for (int j=i+1;j<len;j++){
                if (less(a[j],a[i])){
                    exchange(a, i, j);
                }
            }
        }
    }
    private static boolean less(Comparable v, Comparable w){
        return v.compareTo(w) < 0;
    }
    private static void exchange(Comparable[] a, int i, int j){
        Comparable swap = a[i];
        a[i] = a[j];
        a[j] = swap;
    }
    public static boolean isSorted(Comparable[] a){
        for (int i=1;i<a.length;i++){
            if (less(a[i],a[i-1])){
                return false;
            }
        }
        return true;
    }
}
    1. if the array is sorted, it does not matter to selection sort

    public class Shell {
    //  3x+1 increment sequence
    public static void sort(Comparable[] a){
        int len = a.length;
        int h = 1;
        // why h < len/3?
        while(h<len/3){
            h = 3*h+1;
        }
        while(h>0){
            for (int i=0;i<len;i++){
                // there, should j >= h instead of j > 0;
                for (int j=i;j>=h && less(a[j], a[j-h]);j-=h){
                    exchange(a, j, j-h);
                }
            }
            h = h - 3;
